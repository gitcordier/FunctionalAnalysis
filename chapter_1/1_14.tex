\textit{Put $K =[0, 1]$ and define $\D_K$ as in Section 1.46. Show that the following three families of seminorms %
(where $n = 0, 1, 2, \dots$) define the same topology on $\D_K$. If $D = d/dx$: 
%
\begin{enumerate}
\item{
  $\norm*[\infty]{D^n f} = \sup\set{\magnitude*{D^n f(x)}}{\minus \infty< x< \infty}$
}
\item{
  $\norm*[1]{D^n f} = \int_0^1 \magnitude*{D^n f(x)} \diff{x}$
}
\item{
  $\norm*[2]{D^n f} = \singleton{\int_0^1 \magnitude*{D^n f(x)}^2 \diff{x}}^{1/2}.$
}
\end{enumerate}}
%
\begin{proof}%
The vector space $\Continuous_K = C(\R)\cap \set{u}{\supp{u} \subset K}$, which contains $\D_K$, is equipped with the inner %
product $\bra{u}\ket{v} = \int_0^1 u \bar{v}$, so that $\bra{u}\ket{u} = \norm*[2]{u}^2$. The Cauchy-Schwarz inequality with %
$u=1$ and $v = \magnitude*{D^n f}$ gives
%
\begin{equation}
  \int_0^1 1 \cdot \magnitude*{D^n f} \leq \norm*[2]{1} \cdot \norm*[2]{D^n f};
\end{equation}
%
see Theorem \citeFA{12.2}. Since $K$ has length $1$, we obtain %
%
\begin{equation}\label{1.14:inequalities 1}
  \norm*[1]{D^n f} \leq \norm*[2]{D^n f} \leq \norm*[\infty]{D^n f}.
\end{equation}
%
Moreover, having $f$ supported by $K$ yields
%
\begin{equation}
  \magnitude*{D^n f(x)} = \magnitude*{\int_{0}^x D^{n+1}f} 
  \leq \int_{0}^x \magnitude*{D^{n+1}f} 
  \leq \norm*[1]{D^{n+1}f}.
\end{equation}
%
Taking the supremum of $\magnitude*{D^n f(x)}$ over all $x$ strengthens \eqref{1.14:inequalities 1}, as follows:
%
\begin{equation}\label{inequalities 3}
  \norm*[1]{D^n f}
  \leq \norm*[2]{D^n f}
  \leq \norm*[\infty]{D^n f} 
  \leq \norm*[1]{D^{n+1} f} 
  \leq \cdots .
\end{equation}
%
Finally, set
%
\begin{equation}
  V^{[i]}_n \Def \set{f}{f \in \D_K, \norm*[i]{D^n f} < 2^{\minus n}} 
\end{equation}
%
for each seminorm $i$, so that (\ref{inequalities 3}) is reflected in the following neighborhood inclusions:
%
\begin{equation}\label{1.14:inclusions}
  V^{[1]}_n \supseteq V^{[2]}_n 
  \supseteq V^{[\infty]}_n 
  \supseteq V^{[1]}_{n+1} 
  \supseteq \cdots .
\end{equation}
%
The inclusions $V^{[i]}_n\supseteq V^{[i]}_{n+1}$ turn $\set{V^{[i]}_n}{\naturals{n}}$ into a local base of a topology %
$\tau_i$. The entire chain \eqref{1.14:inclusions} forces %
%
\begin{equation}
  \tau_1 \subset \tau_2 \subset \tau_\infty \subset \tau_1, 
\end{equation}
%
which completes the argument.
\end{proof}
