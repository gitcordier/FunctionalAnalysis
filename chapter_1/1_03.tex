\textit{
Let be X as topological vector space. All sets mentioned below are understood to be the subsets of X. Prove the following %
statements:
\begin{enumerate}
\item The convex hull of every open set is open.
\item If X is locally convex then the convex hull of every bounded set is bounded.
\item If A and B are bounded, so is A+B.
\item If A and B are compact, so is A+B.
\item If A is compact and B is closed, then A+B is closed.
\item The sum of two closed sets may fail to be closed.
\end{enumerate}
}
%: (a)
\begin{proof}%
\textbf{Proof of (a).} Let $A$ be open, so that 
%
\begin{equation}
  \co(A) = \bigcup \set[\big]{\co(V_1, \dots, V_n)}{{V_i \subset A, \text{ $V_i$ open, $n$ is arbitrary}}},
\end{equation}
%
where $\co(V_1, \dots, V_n)$ is the union of all convex combinations $t_1 V_1 +\cdots + t_n V_n$ %
($t_1 + \cdots + t_n = 1$, $t_i > 0$). A straightforward induction on $n$ proves that every mapping %
$(x_1, \dots, x_n) \mapsto t_1 x_1 + \cdots + t_n x_n$ is open. The base cases $n=1, 2$ are provided by \citeFA{1.7}. %
As a result, every $t_1 V_1 +\cdots + t_n V_n$ is an open subset of $\co(V_1,\dots,V_n)$. Therefore, $\co(V_1,\dots,V_n)$ is %
open. Property (a) follows, since $\co(A)$ is the union of all $\co(V_1,\dots, V_n)$.
%:(b)
\paragraph{Proof of (b).} Given a bounded $E$, let $V$ be any neighborhood of $0$. By \citeFA{1.14 (b)}, $V$ contains a convex 
neighborhood $W$ of $0$. It follows that there exists $s > 0$ such that
%
\begin{equation}
  E \subset tW \subset tV \qquad (t > s). 
\end{equation}
%
Hence %
%
\begin{equation}
  \co(E) \subset \co(tW) = t\co(W) = t W \subset tV, 
\end{equation}
%
which completes the proof. %
%
%:(c)
\paragraph{Proof of (c).} For a fixed neighborhood $V$ of the origin, the continuity of $+$, combined with \citeFA{1.14}, %
provides a balanced open $U$ such that %
%
\begin{equation}\label{1.3.c:U+U-subset-of-V}
  U + U\subset V. 
\end{equation}
%
By boundedness, $A \subset r U$ for some $r> 0$, and $B \subset s U$ for some $s>0$. For $t > \max(r,s)$, the balancedness of %
$U$ gives 
%
\begin{align}
  A+B & \subset  r U + s U \\
  & \subset t U + t U \\
  &\subset t V, && \bigl(\text{from }\eqref{1.3.c:U+U-subset-of-V}\bigr)
\end{align}
%
which concludes the proof. %
%
%:(d)
\paragraph{Proof of (d).} Since $A \times B$ is compact and the mapping
%
\begin{align}
  +: A \times B &\to A + B \\\nonumber
  (a,b) &\mapsto a + b 
\end{align}
%
is continuous and surjective, the sum $A + B$ is compact. 
%
%:(e)
\paragraph{Proof of (e).} Assume that $A\neq\emptyset$ and $B\neq\emptyset$, since otherwise the result is trivial. %
Pick $c \in X \setminus (A+B)$. It suffices to show that $c\in X\setminus\closure{A+B}$. To this end, let $a$ range over $A$. %
The set $a+B$ is closed; see \citeFA{1.7}. Since $(a+B) \cap \singleton{c} = \emptyset$, it follows from \citeFA{1.10} that %
there exists a neighborhood $V=V(a)$ of the origin such that %
%
\begin{equation}\label{1.3.e:separation}
  (a+B + V) \cap (c+V) = \emptyset.
\end{equation}
%
As a compact subset, $A$ is covered by finitely many $a+V$, say $a_1 + V_1, \dots, a_N + V_N$, whose union is $U$. Hence 
%
\begin{equation}\label{1.3.e:U+B-encloses-A+B}
  A+B \subset U + B.
\end{equation}
%
Now choose
%
\begin{equation}
  W \Def \bigcap_{i=1}^N V_i
\end{equation}
%
so that 
%
\begin{equation}
  (a_i + V_i + B) \cap (c + W) \overset{\eqref{1.3.e:separation}}{=} \emptyset 
\end{equation}
%
for every $a_i + V_i$. Thus, $c \notin \closure{U + B}$. Finally, \eqref{1.3.e:U+B-encloses-A+B} establishes that %
$c \notin \closure{A+B}$. This completes the proof. 
%
%:(f)
\paragraph{Proof of (f).} We provide a counterexample. Let $f: \R \to \R$ be continuous such that 
%
\begin{enumerate}[label=(\roman*)]
  \item $f(x) + f(\minus x) \neq 0$ \qquad ($x \in \R$),
  \item $f$ vanishes at infinity. 
\end{enumerate} 
% 
Condition (i) is satisfied when $f(x) > 0 $ for all $x$. For simplicity, choose $f$ to be strictly positive and even, \eg the %
\emph{bell curve} $f(x)= \exp(\minus x^2)$. Other canonical choices are $f(x)=2^{\minus\abs{x}}$, $f(x)=\frac{1}{1+\abs{x}}$, % 
and so on. As a continuous function, $f$ has a closed graph $G$; see \citeFA{2.14}. Observe that $(0, 0)\notin G+G$ because%
$(0, 0) \in G+G$ implies
% 
\begin{equation}
  (0, 0) =  \bigl(x_0 + (\minus x_0), \, \underbrace{f(x_0)+ f(\minus x_0)}_{> 0}\,\bigr) % 
\end{equation}
%
for some $x_0$, which contradicts (i). However, 
%
\begin{equation}
  \bigl(0, f(n) + f(\minus n)\bigr) \in G + G \qquad (\counting{n}).
\end{equation}
%
%
Finally, it follows from (ii) that % 
%
\begin{equation}
  \bigl(0, f(n)+f(\minus n)\bigr) \tendsto{n}{\infty} (0, 0). 
\end{equation}
%
We have thus constructed a sequence in $G+G$ that converges to $(0,0) \notin G+G$. The proof is complete. 
%
\end{proof}
%
\begin{lemma}[Corollary to (e)]
If $B = \closure{S}$, then %
%
\begin{equation}
  A+B \subset \closure{A+S} \subset \closure{A+B} = A + B
\end{equation}
%
The special case where $A = \{x\}$ and $B=X$ applies to Exercise 15 in chapter 2.
\end{lemma}
%
\begin{proof}
The result directly follows from \citeFA{1.13 (b)}, since $A$ is closed; see \citeFA{1.12}. 
\end{proof}
