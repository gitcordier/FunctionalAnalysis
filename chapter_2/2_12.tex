\textit{%
Let X be the normed space of all real polynomials in one variable, with %
%
\begin{equation*}
  \norm*{f} = \int_0^1 \magnitude*{f(t)} \diff{t}.
\end{equation*}
%
Put $B(f, g) = \int _0^1 f(t)g(t) \diff{t}$, and show that $B$ is a bilinear functional on $X\times X$ which is separately %
continuous but is not continuous.}
% 
\begin{proof}
Observe that %
%
\begin{equation}\label{2.12:magnitude-of-B}
  \magnitude*{B(f, g)} \leq \max_{[0,1]} \magnitude*{g} \norm*{f},
\end{equation}
%
which shows that the partial mapping $f\mapsto B(f,g)$ is continuous; see \citeFA{1.18}. By symmetry, the partial mapping %
$g\mapsto B(f,g)$ is likewise continuous. This concludes the first part. We now prove, however, that $B$ is not continuous. %
To this end, suppose, for a contradiction, that $B$ is continuous. Applying [2.9] with $X=Y=\R[T]$ and $Z=\R$ (completeness %
is not needed) establishes that 
%
\begin{equation}
  \magnitude*{B(f, f)} \leq M \norm*{f}^2 \qquad (f \in X)
\end{equation}
%
for some $M < \infty$. Therefore, evaluating at
%
\begin{equation}
    f_n(T)\Def \sqrt{n} \, T^{n} \qquad (\counting{n})
\end{equation}
%
yields the desired contradiction:
% 
\begin{equation}
  \frac{1}{3} \leq B(f_n, f_n) = \frac{n}{2n+1} \leq M \norm*{f_n}^2 = \frac{M n}{(n+1)^2} \tendsto{n}{\infty} 0.
\end{equation}
%
Alternatively, without relying on [2.9], we observe that $\norm*{f_n}\to 0$, which implies that $(f_n, f_n)$ tends to $(0,0)$ %
in the product space $X \times X$. In contrast, $B(f_n, f_n) \to 1/2 \neq B(0,0)=0$. Thus, $B$ is not sequentially continuous %
and, consequently, not continuous. This completes the proof.
\end{proof}
