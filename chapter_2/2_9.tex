\textit{
Suppose $X,Y,Z$ are Banach spaces and 
%
\begin{align*}
  B:X\times Y \to Z
\end{align*}
%
is bilinear and continuous. Prove that there exists $M<\infty$ such that 
%
\begin{align*}
  \norm{B(x, y)}
  \leq 
  M \norm{x}\norm{y} \quad (x\in X, y\in Y).
\end{align*}
%
Is completeness needed here?}
%
\begin{proof}%
% % % % % % % 
% FIRST PART 
% % % % % % % 
Completeness is not required. For the proof, we only assume that $X$, $Y$, and $Z$ are normed spaces. Since $B$ is continuous, 
there exists $r > 0$ such that 
%
\begin{equation}\label{bound for B}
  \norm{B(x, y)} < 1 
\end{equation}
%
whenever $\bigl(\norm{x} + \norm{y} < r \bigr)$. Therefore, any $0 < s < r$ yields%
%
\begin{align}
  \norm{B(x, y)} = \frac{4\norm{x}\norm{y}}{s^2} 
    \left\| B \left(\frac{s}{2}\cdot \frac{x}{\norm{x}}, \frac{s}{2} \cdot \frac{y}{\norm{y}} \right)\right\|
  < \frac{4 \norm{x} \norm{y}}{s^2} \qquad (x \neq 0, y \neq 0)
\end{align}
because $ \norm{\frac{s}{2}\cdot \frac{x}{\norm{x}}} + \norm{\frac{s}{2} \cdot \frac{y}{\norm{y}}} = s < r$. This establishes %
the existence of $M=4/r^2$, since
%
\begin{align}
    \norm{B(x, y)} <  %
    \frac{\!\!\!4\!}{\,s^{\,2}}\norm{x}\norm{y} 
    \tendsto{s}{r}
    \frac{\!\!\!4\!}{\,r^{\,2}}\norm{x}\norm{y}
    \quad \bigl((x, y) \in X \times Y\bigr). 
\end{align}
% 
% % % % % % % %
% SECOND PART 
% % % % % % % %
Consider the concrete example where $X = Y = Z = C_c(\R)$ equipped with the supremum norm. The existence of $M=1$ follows from
%
\begin{equation}
  \norm[\infty]{fg} \leq \norm[\infty]{f} \cdot \norm[\infty]{g}. %
\end{equation}
%
However, $C_c(\R)$ is not complete; see \citebook{5.4.4}{AnalyseIII}. We prove that the bilinear product %
%
\begin{align}
  B: C_c(\R)^{2} & \to  C_c(\R) \\
  (f, g) & \mapsto f \cdot g \nonumber
\end{align}
%
is nevertheless continuous. To do so, we start with $r=r(f, g, \epsilon)$ as follows:
%
\begin{equation}
  r \triangleq \frac{\epsilon}{1 + \norm[\infty]{f} + \norm[\infty]{g}} < 1 \qquad (0 < \epsilon < 1).
\end{equation}
%
Next, we choose $(u, v) \in C_c(\R)^2$ with  
%
\begin{equation}
  \norm[\infty]{f-u} + \norm[\infty]{g-v} < r, %
\end{equation}
%
so that $\norm[\infty]{fg - uv} < \epsilon$. Explicitly, we have
%
\begin{align}
\norm[\infty]{fg - uv} & = \norm[\infty]{(f-u) \cdot g + u \cdot (g-v)} &\\
  & \leq \norm[\infty]{f-u} \cdot \norm[\infty]{g} + \norm[\infty]{u} \cdot \norm[\infty]{g-v} &\\
  & < r \cdot \norm[\infty]{g} + \left(r + \norm[\infty]{f}\right) \cdot r & 
    \left(\text{using $\norm[\infty]{u} \leq r + \norm[\infty]{f}$}\right)\\
  & < r \cdot \left(r + \norm[\infty]{f} + \norm[\infty]{g} \right)  & \\
  & < \epsilon & \left(\text{because $r < 1$}\right).
\end{align}
%
Since $\epsilon$ is arbitrary, this establishes that $B$ is continuous.
\end{proof}
